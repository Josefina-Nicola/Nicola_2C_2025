114void MostrarDistancia(void *pvParameter){
115 while (1){
116 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
117
118 if(control_on && !control_hold){
119 LcdItsE0803Init();
120 LcdItsE0803Write(distancia);
121

123 sprintf(linea, "%03d cm\r\n", distancia);

125 }
126
127 else if(!control_on){

155void app_main(void){
156
157 LedsInit();
158 SwitchesInit();
159 HcSr04Init(GPIO_3, GPIO_2);
160 LcdItsE0803Init();
161
162 // Inicialización y configuración del puerto serie
163 serial_config_t mi_serial;
164 mi_serial.port = UART_PC;
165 mi_serial.baud_rate = 115200;
166 mi_serial.func_p = EntradaPorTeclado;
167
168 // Inicialización y configuración de la UART
169 UartInit(&mi_serial);
170
171 timer_config_t timer_medicion = {
172 .timer = TIMER_A,
173 .period = MEDICION_PERIODO_US,
174 .func_p = FuncTimer,
175 .param_p = NULL
176 };
177 TimerInit(&timer_medicion);
178
179 xTaskCreate(MedirDistancia, "Medir distancia", 2048, NULL, 5, &medir_task_handle);
180 xTaskCreate(MostrarDistancia, "Mostrar distancia", 4096, NULL, 5, &mostrar_task_handle);
181
182 TimerStart(timer_medicion.timer);
183
184 SwitchActivInt(SWITCH_1, CambiarSwitch1, NULL);
185 SwitchActivInt(SWITCH_2, CambiarSwitch2, NULL);
186}
